## Basic Example

import { List } from "@resembli/le-window"

```jsx live
function FixedSizeListExample() {
  const someItems = [
    ["Apples ðŸŽ", "Keeps the doctor away"],
    ["Pear ðŸ", "People think it's an apple"],
    ["Cherries ðŸ’", "Not berries"],
    ["Strawberry ðŸ“", "The best berries"],
    ["Burger ðŸ”", "Nice any time of the day"],
    ["Pizza ðŸ•", "Cheesey goodness"],
    ["Meat ðŸ–", "To stay strong"],
    ["Chicken Leg ðŸ—", "I hope the chicken is alright"],
  ]

  const data = Array(1000)
    .fill(0)
    .map((_, i) => {
      const index = i % someItems.length
      const props = {
        item: someItems[index][0],
        desc: someItems[index][1],
        price: (Math.random() * 1000).toFixed(2),
      }
      return { props }
    })

  const ItemComponent = (props) => {
    return (
      <div
        className="small-font-for-mobile"
        style={{
          display: "inline-grid",
          paddingLeft: "1rem",
          paddingRight: "1rem",
          gridTemplateColumns: "1fr 2fr auto",
          width: "100%",
          alignItems: "center",
          borderBottom: "1px solid grey",
          height: "100%",
        }}
      >
        <div>{props.item}</div>
        <div>{props.desc}</div>
        <div style={{ color: "green", fontWeight: "bold" }}>${props.price}</div>
      </div>
    )
  }

  return (
    <div style={{ height: 250, border: "1px solid grey" }}>
      <List data={data} defaultRowHeight={50} ItemComponent={ItemComponent} />
    </div>
  )
}
```
