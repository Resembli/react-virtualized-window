## Basic Example

import { List } from "@resembli/le-window"

```jsx live
function FixedSizeListExample() {
  const someItems = [
    ["Apples ðŸŽ", "Keeps the doctor away", 200],
    ["Pear ðŸ", "People think it's an apple", 300],
    ["Cherries ðŸ’", "Not berries", 400],
    ["Strawberry ðŸ“", "The best berries", 500],
    ["Burger ðŸ”", "Nice any time of the day", 600],
    ["Pizza ðŸ•", "Cheesey goodness", 700],
    ["Meat ðŸ–", "To stay strong", 800],
    ["Chicken Leg ðŸ—", "I hope the chicken is alright", 900],
  ]

  const data = Array(1000)
    .fill(0)
    .map((_, i) => {
      const index = i % someItems.length
      const props = {
        item: someItems[index][0],
        desc: someItems[index][1],
        price: someItems[index][2],
      }
      return { props }
    })

  const ItemComponent = (props) => {
    return (
      <div
        className="small-font-for-mobile"
        style={{
          display: "inline-grid",
          paddingLeft: "1rem",
          paddingRight: "1rem",
          gridTemplateColumns: "1fr 2fr auto",
          width: "100%",
          alignItems: "center",
          borderBottom: "1px solid grey",
          height: "100%",
        }}
      >
        <div>{props.item}</div>
        <div>{props.desc}</div>
        <div style={{ color: "green", fontWeight: "bold" }}>${props.price}</div>
      </div>
    )
  }

  return (
    <div style={{ height: 250, border: "1px solid grey" }}>
      <List data={data} defaultRowHeight={50} ItemComponent={ItemComponent} />
    </div>
  )
}
```

## Asynchrono List Items with React Suspense Example

```jsx live
function AsyncFixedSizeListExample() {
  const fakeRead = () =>
    new Promise((resolve) =>
      setTimeout(() => {
        resolve()
      }, 1000),
    )

  const AsyncItem = () => {
    fakeRead()

    return <div>I have loaded</div>
  }

  const SuspenseItem = () => {
    return (
      <Suspense fallback="Loading...">
        <AsyncItem />
      </Suspense>
    )
  }

  const data = Array(1000)
    .fill(0)
    .map((_, i) => ({ props: {} }))

  return (
    <div style={{ height: 250, border: "1px solid grey" }}>
      <List data={data} defaultRowHeight={50} ItemComponent={SuspenseItem} />
    </div>
  )
}
```
